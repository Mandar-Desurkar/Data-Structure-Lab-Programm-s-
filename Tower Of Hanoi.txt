#include <stdio.h>
//TOH-Tower Of Hanoi//
void TOH(int,char,char,char);

void main()
{   int num;
    printf("Enter the number of Dis you want to insert:\n");
    scanf("%d",&num);
    TOH(num,'A','B','C');
}

void TOH(int num,char X,char Y,char Z)
{
    if (num>0)
    {
        TOH(num-1,X,Z,Y);
        printf("\n%c---->%c",X,Z);
        TOH(num-1,Y,X,Z);
    }

}
