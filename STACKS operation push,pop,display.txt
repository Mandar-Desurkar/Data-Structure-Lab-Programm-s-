#include <stdio.h>
#include<stdio.h>
#define N 5
int stack[N];
int top=-1;

void push();
void pop();
void display();




int main()
{   int choice;
    int option=1;
   
   
    while(option)
    {
        printf("\n Enter the Choice:\n 1.PUSH\n 2.POP\n 3.DISPLAY\n 4.EXIT\n");
        scanf("%d",&choice);
        switch(choice)
        {
            case 1:
            {
                push();
                break;
            }
            case 2:
            {
                pop();
                break;
            }
            case 3:
            {
                display();
                break;
            }
            case 4:
            {
                option=0;
                printf("\n EXIT POINT ");
                break;
                
            }
            default:
            {
                printf ("\nPlease Enter a Valid Choice(1/2/3/4)");
            }
                
        }
    }
    
}


void push()
{
    int x;
    printf("Enter the Data in STACKS:");
    scanf("%d",&x);
    if(top==N-1)
    {
        printf("\nSTACK is over flow");
    }
    
    else
    {   top++;
        stack[top]=x;
    }
}


void pop()
{   int item;

    if(top==-1)
    {
        printf("\n Stack is under flow");
    }
    else
    {   item=stack[top];
        top--;
        printf("\n The popped elements is %d",item);
    }
}


void display()
{   int i;
    printf("The Items in STACK are:\n");
        for(i=top; i>=0; i--)
        { 
            printf("%d\n",stack[i]);
          
        }
}
